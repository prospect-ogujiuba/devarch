2. Replace the permissive `CheckOrigin` with origin validation:
   ```go
   CheckOrigin: func(r *http.Request) bool {
       // Wildcard: allow all
       if len(allowedOrigins) == 0 || (len(allowedOrigins) == 1 && allowedOrigins[0] == "*") {
           return true
       }
       origin := r.Header.Get("Origin")
       if origin == "" {
           // No Origin header (e.g. non-browser clients) — allow
           return true
       }
       for _, allowed := range allowedOrigins {
           if origin == allowed {
               return true
           }
       }
       log.Printf("websocket: rejected connection from origin %q", origin)
       return false
   },
   ```

3. In `api/internal/api/routes.go`, update the `NewWebSocketHandler` call to pass the `allowedOrigins` slice:
   ```go
   wsHandler := handlers.NewWebSocketHandler(syncManager, allowedOrigins)
   ```
