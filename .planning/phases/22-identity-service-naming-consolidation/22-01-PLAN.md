```go
package identity

func NetworkName(stackName string) string
// Returns fmt.Sprintf("devarch-%s-net", stackName)

func ContainerName(stackName, instanceName string) string
// Returns fmt.Sprintf("devarch-%s-%s", stackName, instanceName)

func ResolveNetworkName(stackName string, customNetworkName *string) string
// If customNetworkName non-nil and non-empty, return it. Otherwise return NetworkName(stackName).

func ResolveContainerName(stackName, instanceName string, customContainerName *string) string
// If customContainerName non-nil and non-empty, return it. Otherwise return ContainerName(stackName, instanceName).

func ExportFileName(stackName string) string
// Returns fmt.Sprintf("%s-devarch.yml", stackName)

func LockFileName(stackName string) string
// Returns fmt.Sprintf("%s-devarch.lock", stackName)

func ExtractInstanceName(stackName, containerName string) string
// Strips "devarch-{stackName}-" prefix. Returns "" if prefix doesn't match.
```

These are pure functions — no DB dependency needed. Callers query DB for custom names and pass them in (transport-agnostic per Phase 21 decision).
